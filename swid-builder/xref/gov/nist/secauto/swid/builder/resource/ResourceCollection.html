27  package gov.nist.secauto.swid.builder.resource;
28  
29  import java.security.MessageDigest;
30  import java.security.NoSuchAlgorithmException;
31  import java.util.Collection;
32  import java.util.Collections;
33  import java.util.HashMap;
34  import java.util.Map;
35  import java.util.TreeMap;
36  
37  public class ResourceCollection {
38    Map<String, ResourcePath> pathToResourcePathMap;
39    Map<ResourcePath, ResourceEntry> resources;
40  
41    /**
42     * Construct a new empty resource collection.
43     */
44    public ResourceCollection() {
45      super();
46      pathToResourcePathMap = new HashMap<>();
47      resources = new TreeMap<>();
48    }
49  
50    /**
51     * Add a resource to the resource collection.
52     * 
53     * @param resource
54     *          the resource to add
55     */
56    public void addResource(ResourceEntry resource) {
57      ResourcePathlder/resource/ResourcePath.html#ResourcePath">ResourcePath key = new ResourcePath(resource.getPath());
58      pathToResourcePathMap.put(key.getPath(), key);
59      resources.put(key, resource);
60    }
61  
62    public Collection<ResourceEntry> getResources() {
63      return Collections.unmodifiableCollection(resources.values());
64    }
65  
66    /**
67     * Lookup a resource with a specific path.
68     * 
69     * @param path
70     *          the path to lookup a resource for
71     * @return a resource record or <code>null</code> if no record was found matching the path
72     */
73    public ResourceEntry getResource(String path) {
74      ResourcePath key = pathToResourcePathMap.get(path);
75      ResourceEntry retval;
76      if (key != null) {
77        retval = resources.get(key);
78      } else {
79        retval = null;
80      }
81      return retval;
82    }
83  
84    /**
85     * Retrieve a digest of all of the resources in the collection. This can be used to check if two
86     * resource collections are equal.
87     * 
88     * @param algorithm
89     *          the hash function to use to generate the digest of the collection
90     * @return a digest based on the provided function
91     * @throws NoSuchAlgorithmException
92     *           if the requested has function is unsupported
93     */
94    public byte[] getMessageDigest(HashAlgorithm algorithm) throws NoSuchAlgorithmException {
95      MessageDigest digest = MessageDigest.getInstance(algorithm.getName());
96  
97      for (ResourceEntry data : resources.values()) {
98        digest.update(HashUtils.toArray(data.getDigestValue(HashAlgorithm.SHA_512)));
99      }
100     return digest.digest();
101   }
102 }
