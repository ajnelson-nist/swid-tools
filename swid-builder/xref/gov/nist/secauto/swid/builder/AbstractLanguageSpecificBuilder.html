27  package gov.nist.secauto.swid.builder;
28  
29  import gov.nist.secauto.swid.builder.util.Util;
30  
31  public abstract class AbstractLanguageSpecificBuilder<E extends AbstractLanguageSpecificBuilder<E>>
32      extends AbstractBuilder {
33  
34    private String language;
35  
36    public AbstractLanguageSpecificBuilder() {
37      reset();
38    }
39  
40    @Override
41    public void reset() {
42      this.language = null;
43    }
44  
45    public String getLanguage() {
46      return language;
47    }
48  
49    /**
50     * Sets the to-be-built tag data element's language to the provided language value. It is expected
51     * that the language code conforms with ISO 639 language codes and ISO 3166â€‘1 country codes.
52     * 
53     * @see <a href="https://tools.ietf.org/html/rfc5646">RFC5646</a>
54     * 
55     * @param lang
56     *          the language value
57     * @return the same builder instance
58     */
59    @SuppressWarnings("unchecked")
60    public E language(String lang) {
61      Util.requireNonEmpty("language", lang);
62      Util.requirePatternMatch(SWIDConstants.PATTERN_LANGUAGE, lang, "language");
63      this.language = lang;
64      return (E) this;
65    }
66  
67    @Override
68    public void validate() throws ValidationException {
69    }
70  }
