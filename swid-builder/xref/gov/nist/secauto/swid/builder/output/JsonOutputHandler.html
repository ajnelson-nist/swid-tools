103   private static String lookupFieldName(long fieldId) {
104     return fieldIdToNameMap.get(fieldId);
105   }
106 
107   public JsonOutputHandler() {
108     this(new JsonFactory());
109   }
110 
111   public JsonOutputHandler(JsonFactory jsonFactory) {
112     super(jsonFactory);
113   }
114 
115   @Override
116   protected void writeField(JsonGenerator generator, long fieldId) throws IOException {
117     generator.writeFieldName(lookupFieldName(fieldId));
118   }
119 
120   @Override
121   protected void writeRole(JsonGenerator generator, Role role) throws IOException {
122     generator.writeString(role.getName());
123   }
124 
125   @Override
126   protected void writeVersionScheme(JsonGenerator generator, VersionScheme versionScheme) throws IOException {
127     generator.writeString(versionScheme.getName());
128   }
