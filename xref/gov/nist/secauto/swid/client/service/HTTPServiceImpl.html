138   /**
139    * Execute insert or update http request
140    * 
141    * @param httpClient
142    * @param swidDataFilePath
143    * @param action
144    * @param token
145    * @return
146    * @throws IOException
147    * @throws TransformerException
148    */
149   private String callEndPoint(CloseableHttpClient httpClient, List<String> swidFiles, Action actionType, String token,
150       TagType type) throws IOException, TransformerException {
151     PayloadBuilderService payloadBuilder = new PayloadBuilderServiceImpl();
152     String responseString = "";
153     String endPointURL = "";
154 
155     endPointURL = actionType.fetchEndpoint();
156     LOG.info("Calling SWID " + actionType.action + ": " + endPointURL);
157     String payload = payloadBuilder.buildPayload(swidFiles, type);
158     StringEntity entity = new StringEntity(payload);
159 
160     // HttpPost
161     HttpUriRequest httpRequest = new HttpPost(endPointURL);
162     if (actionType.equals(Action.insert)) {
163       httpRequest = new HttpPost(endPointURL);
164       ((HttpPost) httpRequest).setEntity(entity);
165     } else if (actionType.equals(Action.update)) {
166       httpRequest = new HttpPut(endPointURL);
167       ((HttpPut) httpRequest).setEntity(entity);
168     }
169 
170     httpRequest.setHeader("Accept", "application/xml");
171     httpRequest.setHeader("Content-type", "application/xml");
172     httpRequest.setHeader("Authorization", "Bearer " + token);
173 
174     LOG.info("SWID request payload *****************************************\n" + prettyFormat(payload, 2));
175 
176     HttpResponse httpResponse = httpClient.execute(httpRequest);
177     LOG.info("Response Status :" + httpResponse.getStatusLine());
178     BufferedReader rd = new BufferedReader(new InputStreamReader(httpResponse.getEntity().getContent()));
179     String line = "";
180     while ((line = rd.readLine()) != null) {
181       responseString += line;
182     }
183 
184     return responseString;
185   }

187   /**
188    * Format XML response for human readability
189    * 
190    * @param input
191    *          input string to format
192    * @param indent
193    *          the number of space to indext
194    * @return the formatted string
195    * @throws TransformerException
196    *           if transformation error occurs
197    */
198   public static String prettyFormat(String input, int indent) throws TransformerException {
199 
200     if (input == null || input.isEmpty()) {
201       return null;
202     }
203     Source xmlInput = new StreamSource(new StringReader(input));
204     StringWriter stringWriter = new StringWriter();
205     StreamResult xmlOutput = new StreamResult(stringWriter);
206     TransformerFactory transformerFactory = TransformerFactory.newInstance();
207     // transformerFactory.setAttribute("indent-number", indent);
208     Transformer transformer = transformerFactory.newTransformer();
209     transformer.setOutputProperty(OutputKeys.INDENT, "yes");
210     transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
211     transformer.transform(xmlInput, xmlOutput);
212     return xmlOutput.getWriter().toString();
213 
214   }

229   /**
230    * Fetch token if cached
231    * 
232    * @param subjectDN
233    *          the DN of the certificate
234    * @return the cached token string
235    * @throws UnsupportedEncodingException
236    *           if encoding error occurs
237    * @throws IOException
238    *           if error with read or write
239    */
240   public String getCachedToken(String subjectDN) throws UnsupportedEncodingException, IOException {
241     String token = null;
242     String dn = null;
243     String data = null;
244     File file = new File(TOKEN_CACHE_FILENAME);
245     if (file.exists() && file.isFile()) {
246       try {
247         data = new String(Files.readAllBytes(file.toPath()));
248         String[] items = data.split("\n");
249         if (items != null && items.length == 2) {
250           dn = items[0];
251           token = items[1];
252         }
253         if (dn != null && !dn.isEmpty() && !dn.equals(subjectDN)) {
254           throw new JWTException("Cached DN does not match DN of cient certificate provided");
255         }
256         if (token != null) {
257           int index = token.lastIndexOf('.');
258           String withoutSignature = token.substring(0, index + 1);
259           @SuppressWarnings({ "unused", "rawtypes" })
260           Jwt<Header, Claims> untrusted = Jwts.parser().parseClaimsJwt(withoutSignature);
261 
262         } else {
263           throw new JWTException("Cached token is NULL");
264         }
265       } catch (io.jsonwebtoken.ExpiredJwtException exp) {
266         LOG.info("Cached token expired, proceed to attempt re-authentication");
267         return null;
268       } catch (JWTException e) {
269         LOG.info(e.getMessage() + ", proceed to attempt re-authentication");
270         return null;
271       } catch (Exception e) {
272         LOG.info("Cached token is invalid, proceed to attempt re-authentication");
273         return null;
274       }
275 
276     }
277 
278     return token;
279   }

281   /**
282    * Set token cache
283    * 
284    * @param token
285    *          the token to cache
286    * @param subjectDN
287    *          DN of the certificate
288    * @throws IOException
289    *           if read or write error occurs
290    */
291   public void setTokenCache(String token, String subjectDN) throws IOException {
292     FileWriter writer = null;
293 
294     if (token == null) {
295       LOG.info("Unable to cache token, token is null");
296       return;
297     }
298     try {
299 
300       writer = new FileWriter(new File(TOKEN_CACHE_FILENAME));
301       writer.write(subjectDN + "\n" + token);
302       LOG.info("Token cached");
303     } finally {
304       if (writer != null) {
305         writer.close();
306       }
307     }
308   }

310   /**
311    * Get Subject DN from Keystore instance
312    * 
313    * @param ks
314    *          the keystore
315    * @return the string which is the DN of the certificate
316    * @throws KeyStoreException
317    *           if error occurs with keystore loading
318    */
319   public String getKeyStoreSubjectDN(KeyStore ks) throws KeyStoreException {
320     Enumeration<String> enumeration = ks.aliases();
321     String subjectDN = null;
322     while (enumeration.hasMoreElements()) {
323       String alias = (String) enumeration.nextElement();
324 
325       Certificate cert = ks.getCertificate(alias);
326       if (cert instanceof X509Certificate) {
327         X509Certificate x509cert = (X509Certificate) cert;
328 
329         // Get subject
330         Principal principal = x509cert.getSubjectDN();
331         subjectDN = principal.getName();
332         LOG.info("Subject DN from certificate: " + subjectDN);
333       }
334     }
335 
336     return subjectDN;
337 
338   }
